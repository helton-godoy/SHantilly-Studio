188 void serializeWidget(QWidget *widget, QJsonObject &json) override {
189 if (!widget)
190 return;
191 QString t = widget->property("SHantilly_type").toString();

193 if (t == "table") {
194 if (auto *table = qobject_cast<QTableWidget *>(widget)) {
195 QJsonArray headers;

199 json["headers"] = headers;
200
201 // TODO: Rows serialization if needed

208 json["items"] = items;

211 if (auto *list = qobject_cast<QListWidget *>(widget)) {

220 void deserializeWidget(QWidget *widget, const QJsonObject &json) override {
221 if (!widget)
222 return;
223 QString t = widget->property("SHantilly_type").toString();
224
225 if (t == "table") {
226 if (auto *table = qobject_cast<QTableWidget *>(widget)) {
227 if (json.contains("headers")) {
228 QJsonArray arr = json["headers"].toArray();
229 table->setColumnCount(arr.size());
230 QStringList headers;
231 for (const auto &v : arr)
232 headers << v.toString();
233 table->setHorizontalHeaderLabels(headers);
234 }
235 }
236 } else if (t == "combobox") {
237 if (auto *combo = qobject_cast<QComboBox *>(widget)) {
238 if (json.contains("items")) {
239 combo->clear();
240 QJsonArray arr = json["items"].toArray();
241 for (const auto &v : arr)
242 combo->addItem(v.toString());
243 }
244 }
245 } else if (t == "listbox") {
246 if (auto *list = qobject_cast<QListWidget *>(widget)) {
247 if (json.contains("items")) {
248 list->clear();
249 QJsonArray arr = json["items"].toArray();
250 for (const auto &v : arr)
251 list->addItem(v.toString());
252 }
253 }
254 }
