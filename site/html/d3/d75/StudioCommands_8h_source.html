55 DeleteWidgetCommand(Canvas *canvas, QList<QWidget *> widgets,

58 for (QWidget *w : widgets) {
59 WidgetInfo info;
60 info.widget = w;
61 info.parent = w->parentWidget();
62 info.index = (info.parent && info.parent->layout())
63 ? info.parent->layout()->indexOf(w)
64 : -1;
65 m_widgetsInfo.append(info);
66 }
67 setText(QString("Delete %1 items").arg(widgets.size()));
68 }
69
70 void undo() override {
71 for (const auto &info : m_widgetsInfo) {
72 if (!info.parent || info.parent == m_canvas) {
73 m_canvas->addWidget(info.widget);
74 } else {
75 info.widget->setParent(info.parent);
76 if (auto *tabs = qobject_cast<QTabWidget *>(info.parent)) {
77 tabs->insertTab(info.index, info.widget,
78 info.widget->property("title").toString());
79 } else if (info.parent->layout()) {
80 if (auto *box = qobject_cast<QBoxLayout *>(info.parent->layout())) {
81 box->insertWidget(info.index, info.widget);
82 } else {
83 info.parent->layout()->addWidget(info.widget);
84 }
85 }
86 }
87 info.widget->show();
88 }
89 }
90
91 void redo() override {
92 for (const auto &info : m_widgetsInfo) {
93 m_canvas->removeWidget(info.widget);
94 }
95 }

97private:

99 QList<WidgetInfo> m_widgetsInfo;

134

144 newParent ? newParent->objectName() : "Root"));

149 void redo() override { applyMove(m_newParent, m_newIndex); }

151private:
152 void applyMove(QWidget *parent, int index) {
153 if (!parent || !m_widget)
154 return;
155
156 m_widget->setParent(parent);
157 if (parent->layout()) {
158 if (index >= 0) {
159 // Se for um QBoxLayout, podemos inserir
160 if (auto *box = qobject_cast<QBoxLayout *>(parent->layout())) {
161 box->insertWidget(index, m_widget);
162 } else {
163 parent->layout()->addWidget(m_widget);
164 }
165 } else {
166 parent->layout()->addWidget(m_widget);
167 }
168 }
169 m_widget->show();
170 }
171
172 QWidget *m_widget;
173 QWidget *m_oldParent;
174 QWidget *m_newParent;
175 int m_oldIndex;

230 auto *l = new QHBoxLayout(m_container);

261class ChangeLayoutCommand : public QUndoCommand {
262public:
263 ChangeLayoutCommand(QWidget *container, const QString &newType,
264 QUndoCommand *parent = nullptr)
265 : QUndoCommand(parent), m_container(container), m_newType(newType) {
266 if (qobject_cast<QHBoxLayout *>(container->layout())) {
267 m_oldType = "hbox";
268 } else {
269 m_oldType = "vbox";
270 }
271 setText("Change Layout to " + newType);
272 }
273
274 void undo() override { applyLayout(m_oldType); }
275
276 void redo() override { applyLayout(m_newType); }
277
278private:
279 void applyLayout(const QString &type) {
280 if (!m_container)
281 return;
282
283 // 1. Coletar widgets atuais
284 QList<QWidget *> widgets;
285 QLayout *oldLayout = m_container->layout();
286 if (oldLayout) {
287 QLayoutItem *item;
288 while ((item = oldLayout->takeAt(0)) != nullptr) {
289 if (QWidget *w = item->widget()) {
290 widgets.append(w);
291 }
292 delete item;
293 }
294 delete oldLayout;
295 }
296
297 // 2. Criar novo layout
298 QLayout *newLayout = nullptr;
299 if (type == "hbox") {
300 newLayout = new QHBoxLayout(m_container);
301 } else {
302 newLayout = new QVBoxLayout(m_container);
303 }
304 newLayout->setContentsMargins(5, 5, 5, 5);
305
306 // 3. Readicionar widgets
307 for (QWidget *w : widgets) {
308 newLayout->addWidget(w);
309 }
310 }
311
312 QWidget *m_container;
313 QString m_oldType;
314 QString m_newType;
315};
